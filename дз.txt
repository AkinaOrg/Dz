# Запитуємо користувача ввести перше число
num1 = float(input("Введіть перше число: "))

# Запитуємо користувача ввести друге число
num2 = float(input("Введіть друге число: "))

# Створюємо пустий список
my_list = []

# Порівнюємо числа і додаємо більше з них до списку
if num1 > num2:
    my_list.append(num1)
elif num1 < num2:
    my_list.append(num2)
else:
    my_list.append(num1)  # Ви можете вибрати будь-яке з чисел, так як вони рівні

# Виводимо список з найбільшим числом
print("Список з найбільшим числом:", my_list)
